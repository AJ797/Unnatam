\section{Selective Scan with Intracellular Attention}
\label{app:scan-pseudo}

Algorithm~\ref{alg:scan} gives the modified reference selective scan. Lines highlighted in grey are the additions over a vanilla Mamba-1 reference scan.

\begin{algorithm}[h]
\caption{Selective scan with intracellular attention.}
\label{alg:scan}
\begin{algorithmic}[1]
\Require Input $u \in \mathbb{R}^{B \times L \times \dinner}$, time-step $\Delta \in \mathbb{R}^{B \times L \times \dinner}$, state $A \in \mathbb{R}^{\dinner \times \dstate}$, input matrix $B \in \mathbb{R}^{B \times L \times \dstate}$, readout $C \in \mathbb{R}^{B \times L \times \dstate}$, skip $D \in \mathbb{R}^{\dinner}$, intracellular attention module $\mathcal{A}$ (optional)
\State $h \gets \mathbf{0} \in \mathbb{R}^{B \times \dinner \times \dstate}$
\State $\bar A \gets \exp(\Delta_{:,:,:,\text{None}} \odot A)$ \Comment{$(B, L, \dinner, \dstate)$}
\State $\bar B u \gets \Delta_{:,:,:,\text{None}} \odot B_{:,:,\text{None},:} \odot u_{:,:,:,\text{None}}$ \Comment{$(B, L, \dinner, \dstate)$}
\For{$t = 1, \ldots, L$}
    \State $h \gets \bar A_t \odot h + \bar B u_t$
    \If{$\mathcal{A}$ is provided}
        \State $h \gets \mathcal{A}(h)$ \Comment{intracellular attention}
    \EndIf
    \State $y_t \gets \sum_{s} h_{:, :, s} \odot C_{:, t, s}$
\EndFor
\State \Return $y + u \odot D$
\end{algorithmic}
\end{algorithm}
